% ── Folie 3: Textbausteine ────────────────────────────────────────────────────
\begin{frame}{Textbausteine \& Typografie}

    \begin{columns}[T]
        \begin{column}{0.48\textwidth}

            \begin{block}{Normaler Block (\texttt{block})}
                Für neutrale Informationen und Fakten.\\
                \textbf{Fett} · \textit{Kursiv} · \texttt{Monospace}
            \end{block}

            \vspace{0.5em}

            \begin{alertblock}{\textcolor{white}{\textbf{Alert-Block (\texttt{alertblock})}}}
                Für Fazit-Statements und Kernaussagen.\\
                Immer mit \texttt{\textbackslash textcolor\{white\}\{...\}} im Titel.
            \end{alertblock}

            \vspace{0.5em}

            \begin{exampleblock}{Beispiel-Block (\texttt{exampleblock})}
                Für ergänzende Beispiele oder Quellen.
            \end{exampleblock}

        \end{column}
        \begin{column}{0.48\textwidth}

            \textbf{Aufzählungsliste:}
            \begin{itemize}
                \setlength{\itemsep}{0.4em}
                \item Erster Punkt mit \textbf{Hervorhebung}
                \item Zweiter Punkt — \textcolor{bared}{in Rot}
                \item Dritter Punkt — \textcolor{baorange}{in Orange}
                \item Vierter Punkt mit Quelle\footnotemark
            \end{itemize}

            \vspace{0.5em}

            \textbf{Fußnoten-Pattern} (overlay-sicher):
            \begin{itemize}
                \setlength{\itemsep}{0.2em}
                \item[\texttt{→}] \texttt{\textbackslash footnotemark} im Text
                \item[\texttt{→}] \texttt{\textbackslash only<2->\{...\}} für den Text
            \end{itemize}

            \footnotetext[1]{Quelle: Beispiel-Fußnote}

        \end{column}
    \end{columns}

\end{frame}

                \vspace{0.2em}
            \end{block}

            \vspace{0.5em}

            \textbf{Typisches zweispaltiges Layout:}
            \begin{itemize}
                \setlength{\itemsep}{0.3em}
                \item Links: Missverständnis-Block + Bild (Overlay~1)
                \item Rechts: Fakten erscheinen auf Overlay~2
                \item Makro: \texttt{\textbackslash misconceptionslide[bild]\{Titel\}\{...\}}
            \end{itemize}

            \vspace{0.5em}

            \textbf{Format:} 16:9 (\texttt{aspectratio=169}),
            \textbf{Schrift:} 9pt\\
